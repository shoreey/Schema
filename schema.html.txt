    <script>
        const pdfUrl = 'your-schema.pdf'; // Replace with the actual URL of your PDF file
        const pdfViewerDiv = document.getElementById('pdfViewer');
        const downloadPdfLink = document.getElementById('downloadPdf');
        const pdfIframe = document.getElementById('pdfIframe');

        // Function to display the PDF
        function displayPdf() {
            if (pdfIframe.src === '' || pdfIframe.src !== pdfUrl) {
                pdfViewerDiv.innerHTML = '<p class="text-center text-gray-500">Loading PDF...</p>'; // Keep loader
                pdfIframe.style.display = 'none';
                pdfIframe.src = pdfUrl;

                pdfIframe.onload = () => {
                    pdfViewerDiv.innerHTML = ''; // Clear loader
                    pdfIframe.style.display = 'block';
                };
            }
        }
        // Function to set the download link
        function setDownloadLink() {
            downloadPdfLink.href = pdfUrl;
            downloadPdfLink.download = 'CPRD_Schema.pdf'; // suggested filename
        }

        // Call the functions
        displayPdf();
        setDownloadLink();
